6. 파라미터로 임의의 값이 들어있는 정수형 배열을 받아 최대값을 리턴해주는 메소드를 작성해주세요. 
  - (사이즈 3 ~ 10)      
  - (List 사용가능)
package test11;


import java.util.ArrayList;


public class Dron {
	
	static int max(ArrayList<Integer> aa) {
		System.out.println(aa);
		int max = aa.get(0);
		for(int i=1; i<aa.size() ; i++) {
			if(aa.get(i) > max) {
				max = aa.get(i);
			}
		}
		return max;	
	}
	public static void main(String[] args) {
		ArrayList<Integer> al = new ArrayList<Integer>();
		for(int i=0 ; i<10 ; i++) {
			al.add((int)(Math.random()*10)+1);
		}
		System.out.println(max(al));
	}
}


	

7. 파라미터로 String 배열 과 문자열을 받아 해당 배열에 몇번째 방에 해당 문자열이 있는지 리턴하는 메소드를 작성해주세요.
  - (List 사용가능)
 package test11;

import java.util.Arrays;

public class Scv {

	public static int at1(String[] strings, String str) {
		for(int i=0; i<strings.length; i++) {
			if(strings[i].equals(str)) {
				return i;
			}
		}
		return -1;
	}
	public static int at2(String[] strings, String str) {
		return Arrays.asList(strings).indexOf(str);
	}
	public static void main(String[] args) {
		String[] strings = {"hello","world","this","is","java!"};
		
		System.out.println("at1 :" + at1(strings, "is"));
		System.out.println("at2 :" + at2(strings, "c++"));

	}
}
